
public class SwapNumbers
{
public static void main(String args[])
{
        int x = 20;
        int y = 10;
        System.out.println("before swapping x =" +x "\t y= "+y);
        x=x+y;
        y=x-y;
        x=x-y;
        System.out.println("after swapping x =" +x "\t y= "+y);
    }
}
